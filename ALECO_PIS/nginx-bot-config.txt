server {
    listen 80 default_server;
    listen [::]:80 default_server;

    root /var/www/html;
    index index.html index.htm index.nginx-debian.html;
    server_name _;

    # Bot routes - always proxy to Node.js for /advisory/* and /poster/interruption/*
    # Node.js will handle bot detection and serve appropriate content
    location ~ ^/(advisory|poster/interruption)/[0-9]+$ {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header User-Agent $http_user_agent;
    }

    # Default location - serve static files
    location / {
        try_files $uri $uri/ =404;
    }
}
